set(VREEN_DIRECTAUTH_CLIENT_ID "" CACHE TYPE STRING)
set(VREEN_DIRECTAUTH_CLIENT_SECRET "" CACHE TYPE STRING)
set(VREEN_DIRECTAUTH_CLIENT_NAME "noname" CACHE TYPE STRING)

list(APPEND _directauth_defines
    "VREEN_DIRECTAUTH_CLIENT_ID=\"${VREEN_DIRECTAUTH_CLIENT_ID}\""
    "VREEN_DIRECTAUTH_CLIENT_SECRET=\"${VREEN_DIRECTAUTH_CLIENT_SECRET}\""
    "VREEN_DIRECTAUTH_CLIENT_NAME=\"${VREEN_DIRECTAUTH_CLIENT_NAME}\""
)

add_simple_library(vreendirectauth
    STATIC CXX11
    ${INTERNAL_FLAG}
    DEFINE_SYMBOL VK_LIBRARY
    VERSION ${CMAKE_VREEN_VERSION_STRING}
    SOVERSION ${CMAKE_VREEN_VERSION_MAJOR}
    LIBRARIES vreen ${QT_QTWEBKIT_LIBRARY}
    INCLUDES ${PROJECT_BINARY_DIR}/include ../api
    INCLUDE_DIR vreen/auth
    PKGCONFIG_TEMPLATE vreendirectauth.pc.cmake
    DEFINES ${_directauth_defines}
)
